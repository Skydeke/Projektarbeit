\newcommand{\centeredpicture}[4]{
	\begin{figure}[H]
		\centering
		\if\relax\detokenize{#2}\relax
			\includegraphics[width=\textwidth]{#1}
		\else
			\includegraphics[width=#2]{#1}
		\fi
		\caption{#3}
		\label{#4}
	\end{figure}
}
